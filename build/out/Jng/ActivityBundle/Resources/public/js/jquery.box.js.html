  41.  41  *
  42.  42  *  The facebox function can also display an ajax page, an image, or the contents of a div:
  43.  43  *
  44.  44  *    jQuery.facebox({ ajax: 'remote.html' })

  47.  47  *    jQuery.facebox({ image: 'stairs.jpg' }, 'my-groovy-style')
  48.  48  *    jQuery.facebox({ div: '#box' })
  49.  49  *    jQuery.facebox({ div: '#box' }, 'my-groovy-style')
  50.  50  *
  51.  51  *  Want to close the facebox?  Trigger the 'close.facebox' document event:
  52.  52  *

  54.  54  *

  56.  56  *
  57.  57  *    loading.facebox

  60.  60  *    init.facebox
  61.  61  *    afterClose.facebox
  62.  62  *
  63.  63  *  Simply bind a function to any of these hooks:
  64.  64  *
  65.  65  *   $(document).bind('reveal.facebox', function() { ...stuff to do after the facebox and contents are revealed... })

  70.  70     $.facebox.loading(data.settings || [])

  72.  72     if (data.ajax) fillFaceboxFromAjax(data.ajax, klass)

  77.  77   }

  194. 194   function getPageScroll() {
  195. 195     var xScroll, yScroll;
  196. 196     if (self.pageYOffset) {
  197. 197       yScroll = self.pageYOffset;
  198. 198       xScroll = self.pageXOffset;
  199. 199     } else if (document.documentElement && document.documentElement.scrollTop) {  // Explorer 6 Strict
  200. 200       yScroll = document.documentElement.scrollTop;
  201. 201       xScroll = document.documentElement.scrollLeft;
  202. 202     } else if (document.body) {// all other Explorers
  203. 203       yScroll = document.body.scrollTop;
  204. 204       xScroll = document.body.scrollLeft;
  205. 205     }
  206. 206     return new Array(xScroll,yScroll)
  207. 207   }

  237. 237   function fillFaceboxFromHref(href, klass) {

  241. 241       var target = href.replace(url,'')

  243. 243       $.facebox.reveal($(target).html(), klass)
  244. 244 
  245. 245     // image
  246. 246     } else if (href.match($.facebox.settings.imageTypesRegexp)) {
  247. 247       fillFaceboxFromImage(href, klass)
  248. 248     // ajax
  249. 249     } else {
  250. 250       fillFaceboxFromAjax(href, klass)
  251. 251     }
