  52.  52                     case 'w':

  71.  71                 }
  72.  72             }

  116. 116     $.fn.tipsy = function(options) {
  117. 117         
  118. 118         if (options === true) {
  119. 119             return this.data('tipsy');

  135. 135         function enter() {

  139. 139                 tipsy.show();

  141. 141                 setTimeout(function() { if (tipsy.hoverState == 'in') tipsy.show(); }, options.delayIn);

  145. 145         function leave() {

  151. 151                 setTimeout(function() { if (tipsy.hoverState == 'out') tipsy.hide(); }, options.delayOut);
  152. 152             }
  153. 153         };
  154. 154         
  155. 155         if (!options.live) this.each(function() { get(this); });
